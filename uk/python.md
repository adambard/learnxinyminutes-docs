Python –±—É–≤ —Å—Ç–≤–æ—Ä–µ–Ω–∏–π –ì–≤—ñ–¥–æ –≤–∞–Ω –†–æ—Å—Å—É–º–æ–º –Ω–∞ –ø–æ—á–∞—Ç–∫—É 90-—Ö. –ó–∞—Ä–∞–∑ —Ü–µ –æ–¥–Ω–∞ –∑
–Ω–∞–π–ø–æ–ø—É–ª—è—Ä–Ω—ñ—à–∏—Ö –º–æ–≤ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è. –Ø –∑–∞–∫–æ—Ö–∞–≤—Å—è –≤ Python –∑–∞ –π–æ–≥–æ —Å–∏–Ω—Ç–∞–∫—Å–∏—á–Ω—É
—è—Å–Ω—ñ—Å—Ç—å. –¶–µ, –ø–æ —Å—É—Ç—ñ, –≤–∏–∫–æ–Ω—É–≤–∞–Ω–∏–π –ø—Å–µ–≤–¥–æ–∫–æ–¥.

```python
# –û–¥–Ω–æ—Ä—è–¥–∫–æ–≤—ñ –∫–æ–º–µ–Ω—Ç–∞—Ä—ñ –ø–æ—á–∏–Ω–∞—é—Ç—å—Å—è –∑ —Å–∏–º–≤–æ–ª—É —Ä–µ—à—ñ—Ç–∫–∏.

""" –ë–∞–≥–∞—Ç–æ—Ä—è–¥–∫–æ–≤—ñ —Ä—è–¥–∫–∏ –º–æ–∂–Ω–∞ –∑–∞–ø–∏—Å—É–≤–∞—Ç–∏,
    –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—á–∏ —Ç—Ä–∏ –ø–æ–¥–≤—ñ–π–Ω—ñ –ª–∞–ø–∫–∏, —ñ —á–∞—Å—Ç–æ –≤–æ–Ω–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å—Å—è
    —è–∫ –¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü—ñ—è.
"""

####################################################
## 1. –ü—Ä–∏–º—ñ—Ç–∏–≤–Ω—ñ —Ç–∏–ø–∏ –¥–∞–Ω–∏—Ö —ñ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏
####################################################

# –£ –≤–∞—Å —î —á–∏—Å–ª–∞
3  # => 3

# –ú–∞—Ç–µ–º–∞—Ç–∏–∫–∞ –ø—Ä–∞—Ü—é—î —Ç–∞–∫, —è–∫ –≤–∏ –æ—á—ñ–∫—É—î—Ç–µ
1 + 1   # => 2
8 - 1   # => 7
10 * 2  # => 20
35 / 5  # => 7.0

# –¶—ñ–ª–æ—á–∏—Å–µ–ª—å–Ω–µ –¥—ñ–ª–µ–Ω–Ω—è –æ–∫—Ä—É–≥–ª—é—î –≤ –±—ñ–∫ –º—ñ–Ω—É—Å –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ
5 // 3       # => 1
-5 // 3      # => -2
5.0 // 3.0   # => 1.0  # –ø—Ä–∞—Ü—é—î —Ç–∞–∫–æ–∂ –∑ —á–∏—Å–ª–∞–º–∏ –∑ –ø–ª–∞–≤–∞—é—á–æ—é –∫–æ–º–æ—é
-5.0 // 3.0  # => -2.0

# –†–µ–∑—É–ª—å—Ç–∞—Ç –¥—ñ–ª–µ–Ω–Ω—è –∑–∞–≤–∂–¥–∏ —î —á–∏—Å–ª–æ–º –∑ –ø–ª–∞–≤–∞—é—á–æ—é –∫–æ–º–æ—é
10.0 / 3  # => 3.3333333333333335

# –û–ø–µ—Ä–∞—Ü—ñ—è –∑–∞–ª–∏—à–∫—É –≤—ñ–¥ –¥—ñ–ª–µ–Ω–Ω—è (–º–æ–¥—É–ª—å)
7 % 3   # => 1
# i % j –º–∞—î —Ç–æ–π —Å–∞–º–∏–π –∑–Ω–∞–∫, —â–æ –π j, –Ω–∞ –≤—ñ–¥–º—ñ–Ω—É –≤—ñ–¥ C
-7 % 3  # => 2

# –ü—ñ–¥–Ω–µ—Å–µ–Ω–Ω—è –¥–æ —Å—Ç–µ–ø–µ–Ω—è (x**y, x —É —Å—Ç–µ–ø–µ–Ω—ñ y)
2**3  # => 8

# –ü—Ä—ñ–æ—Ä–∏—Ç–µ—Ç –æ–ø–µ—Ä–∞—Ü—ñ–π –º–æ–∂–Ω–∞ –∑–º—ñ–Ω–∏—Ç–∏ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –¥—É–∂–æ–∫
1 + 3 * 2    # => 7
(1 + 3) * 2  # => 8

# –ë—É–ª–µ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è —î –ø—Ä–∏–º—ñ—Ç–∏–≤–∞–º–∏ (–ó–≤–µ—Ä–Ω—ñ—Ç—å —É–≤–∞–≥—É –Ω–∞ –≤–µ–ª–∏–∫—É –ª—ñ—Ç–µ—Ä—É)
True   # => True
False  # => False

# –∑–∞–ø–µ—Ä–µ—á–µ–Ω–Ω—è –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é not
not True   # => False
not False  # => True

# –ë—É–ª–µ–≤—ñ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∏
# –ó–≤–µ—Ä–Ω—ñ—Ç—å —É–≤–∞–≥—É, —â–æ "and" —Ç–∞ "or" —á—É—Ç–ª–∏–≤—ñ –¥–æ —Ä–µ–≥—ñ—Å—Ç—Ä—É
True and False  # => False
False or True   # => True

# True —ñ False –Ω–∞—Å–ø—Ä–∞–≤–¥—ñ —î 1 —ñ 0, –∞–ª–µ –∑ —Ä—ñ–∑–Ω–∏–º–∏ –∫–ª—é—á–æ–≤–∏–º–∏ —Å–ª–æ–≤–∞–º–∏
True + True  # => 2
True * 8     # => 8
False - 5    # => -5

# –û–ø–µ—Ä–∞—Ç–æ—Ä–∏ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è –¥–∏–≤–ª—è—Ç—å—Å—è –Ω–∞ —á–∏—Å–ª–æ–≤–µ –∑–Ω–∞—á–µ–Ω–Ω—è True —ñ False
0 == False   # => True
2 > True     # => True
2 == True    # => False
-5 != False  # => True

# None, 0, —Ç–∞ –ø–æ—Ä–æ–∂–Ω—ñ —Ä—è–¥–∫–∏/—Å–ø–∏—Å–∫–∏/—Å–ª–æ–≤–Ω–∏–∫–∏/–∫–æ—Ä—Ç–µ–∂—ñ/–º–Ω–æ–∂–∏–Ω–∏ –æ—Ü—ñ–Ω—é—é—Ç—å—Å—è —è–∫ False.
# –í—Å—ñ —ñ–Ω—à—ñ –∑–Ω–∞—á–µ–Ω–Ω—è —î True
bool(0)      # => False
bool("")     # => False
bool([])     # => False
bool({})     # => False
bool(())     # => False
bool(set())  # => False
bool(4)      # => True
bool(-6)     # => True

# –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è –ª–æ–≥—ñ—á–Ω–∏—Ö –æ–ø–µ—Ä–∞—Ç–æ—Ä—ñ–≤ –∑ —Ü—ñ–ª–∏–º–∏ —á–∏—Å–ª–∞–º–∏ –ø—Ä–∏–≤–æ–¥–∏—Ç—å —ó—Ö –¥–æ –±—É–ª–µ–≤–∏—Ö –∑–Ω–∞—á–µ–Ω—å
# –¥–ª—è –æ—Ü—ñ–Ω–∫–∏, –∞–ª–µ –ø–æ–≤–µ—Ä—Ç–∞—î—Ç—å—Å—è —ó—Ö–Ω—î –Ω–µ –ø—Ä–∏–≤–µ–¥–µ–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è. –ù–µ –ø–ª—É—Ç–∞–π—Ç–µ –∑ bool(ints)
# —Ç–∞ –ø–æ–±—ñ—Ç–æ–≤–∏–º–∏ and/or (&,|)
bool(0)   # => False
bool(2)   # => True
0 and 2   # => 0
bool(-5)  # => True
bool(2)   # => True
-5 or 0   # => -5

# –†—ñ–≤–Ω—ñ—Å—Ç—å —Ü–µ ==
1 == 1  # => True
2 == 1  # => False

# –ù–µ—Ä—ñ–≤–Ω—ñ—Å—Ç—å —Ü–µ !=
1 != 1  # => False
2 != 1  # => True

# –ë—ñ–ª—å—à–µ –ø–æ—Ä—ñ–≤–Ω—è–Ω—å
1 < 10  # => True
1 > 10  # => False
2 <= 2  # => True
2 >= 2  # => True

# –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞, —á–∏ –∑–Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è –∑–Ω–∞—á–µ–Ω–Ω—è –≤ –¥—ñ–∞–ø–∞–∑–æ–Ω—ñ
1 < 2 and 2 < 3  # => True
2 < 3 and 3 < 2  # => False
# –õ–∞–Ω—Ü—é–∂–∫–∏ —Ä–æ–±–ª—è—Ç—å —Ü–µ –∫—Ä–∞—Å–∏–≤—ñ—à–∏–º
1 < 2 < 3  # => True
2 < 3 < 2  # => False

# (is –ø—Ä–æ—Ç–∏ ==) is –ø–µ—Ä–µ–≤—ñ—Ä—è—î, —á–∏ –ø–æ—Å–∏–ª–∞—é—Ç—å—Å—è –¥–≤—ñ –∑–º—ñ–Ω–Ω—ñ –Ω–∞ –æ–¥–∏–Ω —ñ —Ç–æ–π —Å–∞–º–∏–π –æ–±'—î–∫—Ç,
# –∞–ª–µ == –ø–µ—Ä–µ–≤—ñ—Ä—è—î, —á–∏ –º–∞—é—Ç—å –æ–±'—î–∫—Ç–∏, –Ω–∞ —è–∫—ñ –≤–æ–Ω–∏ –≤–∫–∞–∑—É—é—Ç—å, –æ–¥–Ω–∞–∫–æ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è.
a = [1, 2, 3, 4]  # –í–∫–∞–∑—É—î–º–æ a –Ω–∞ –Ω–æ–≤–∏–π —Å–ø–∏—Å–æ–∫, [1, 2, 3, 4]
b = a             # –í–∫–∞–∑—É—î–º–æ b –Ω–∞ —Ç–µ, –Ω–∞ —â–æ –≤–∫–∞–∑—É—î a
b is a            # => True, a —ñ b –ø–æ—Å–∏–ª–∞—é—Ç—å—Å—è –Ω–∞ –æ–¥–∏–Ω —ñ —Ç–æ–π —Å–∞–º–∏–π –æ–±'—î–∫—Ç
b == a            # => True, –æ–±'—î–∫—Ç–∏ a —ñ b —Ä—ñ–≤–Ω—ñ
b = [1, 2, 3, 4]  # –í–∫–∞–∑—É—î–º–æ b –Ω–∞ –Ω–æ–≤–∏–π —Å–ø–∏—Å–æ–∫, [1, 2, 3, 4]
b is a            # => False, a —ñ b –Ω–µ –ø–æ—Å–∏–ª–∞—é—Ç—å—Å—è –Ω–∞ –æ–¥–∏–Ω —ñ —Ç–æ–π —Å–∞–º–∏–π –æ–±'—î–∫—Ç
b == a            # => True, –æ–±'—î–∫—Ç–∏ a —ñ b —Ä—ñ–≤–Ω—ñ

# –†—è–¥–∫–∏ —Å—Ç–≤–æ—Ä—é—é—Ç—å—Å—è –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é " –∞–±–æ '
"–¶–µ —Ä—è–¥–æ–∫."
'–¶–µ —Ç–∞–∫–æ–∂ —Ä—è–¥–æ–∫.'

# –†—è–¥–∫–∏ —Ç–∞–∫–æ–∂ –º–æ–∂–Ω–∞ –¥–æ–¥–∞–≤–∞—Ç–∏
"Hello " + "world!"  # => "Hello world!"
# –†—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏ (–∞–ª–µ –Ω–µ –∑–º—ñ–Ω–Ω—ñ) –º–æ–∂–Ω–∞ –∫–æ–Ω–∫–∞—Ç–µ–Ω—É–≤–∞—Ç–∏ –±–µ–∑ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è '+'
"Hello " "world!"    # => "Hello world!"

# –†—è–¥–æ–∫ –º–æ–∂–Ω–∞ —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏ —è–∫ —Å–ø–∏—Å–æ–∫ —Å–∏–º–≤–æ–ª—ñ–≤
"Hello world!"[0]  # => 'H'

# –í–∏ –º–æ–∂–µ—Ç–µ –∑–Ω–∞–π—Ç–∏ –¥–æ–≤–∂–∏–Ω—É —Ä—è–¥–∫–∞
len("This is a string")  # => 16

# –ü–æ—á–∏–Ω–∞—é—á–∏ –∑ Python 3.6, –≤–∏ –º–æ–∂–µ—Ç–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ f-—Ä—è–¥–∫–∏ –∞–±–æ —Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω—ñ —Ä—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏.
name = "Reiko"
f"She said her name is {name}."  # => "She said her name is Reiko"
# –ë—É–¥—å-—è–∫–∏–π –≤–∞–ª—ñ–¥–Ω–∏–π –≤–∏—Ä–∞–∑ Python –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ —Ü–∏—Ö –¥—É–∂–æ–∫ –ø–æ–≤–µ—Ä—Ç–∞—î—Ç—å—Å—è –≤ —Ä—è–¥–æ–∫.
f"{name} is {len(name)} characters long."  # => "Reiko is 5 characters long."

# None —î –æ–±'—î–∫—Ç–æ–º
None  # => None

# –ù–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ —Å–∏–º–≤–æ–ª —Ä—ñ–≤–Ω–æ—Å—Ç—ñ "==" –¥–ª—è –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è –æ–±'—î–∫—Ç—ñ–≤ –∑ None
# –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ "is" –∑–∞–º—ñ—Å—Ç—å —Ü—å–æ–≥–æ. –¶–µ –ø–µ—Ä–µ–≤—ñ—Ä—è—î —Ä—ñ–≤–Ω—ñ—Å—Ç—å —ñ–¥–µ–Ω—Ç–∏—á–Ω–æ—Å—Ç—ñ –æ–±'—î–∫—Ç–∞.
"etc" is None  # => False
None is None   # => True

####################################################
## 2. –ó–º—ñ–Ω–Ω—ñ —Ç–∞ –∫–æ–ª–µ–∫—Ü—ñ—ó
####################################################

# Python –º–∞—î —Ñ—É–Ω–∫—Ü—ñ—é print
print("I'm Python. Nice to meet you!")  # => I'm Python. Nice to meet you!

# –ó–∞ –∑–∞–º–æ–≤—á—É–≤–∞–Ω–Ω—è–º —Ñ—É–Ω–∫—Ü—ñ—è print —Ç–∞–∫–æ–∂ –¥—Ä—É–∫—É—î –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –≤ –∫—ñ–Ω—Ü—ñ.
# –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ –Ω–µ–æ–±–æ–≤'—è–∑–∫–æ–≤–∏–π –∞—Ä–≥—É–º–µ–Ω—Ç end, —â–æ–± –∑–º—ñ–Ω–∏—Ç–∏ –∫—ñ–Ω—Ü–µ–≤–∏–π —Ä—è–¥–æ–∫.
print("Hello, World", end="!")  # => Hello, World!

# –ü—Ä–æ—Å—Ç–∏–π —Å–ø–æ—Å—ñ–± –æ—Ç—Ä–∏–º–∞—Ç–∏ –≤—Ö—ñ–¥–Ω—ñ –¥–∞–Ω—ñ –∑ –∫–æ–Ω—Å–æ–ª—ñ
input_string_var = input("Enter some data: ")  # –ü–æ–≤–µ—Ä—Ç–∞—î –¥–∞–Ω—ñ —è–∫ —Ä—è–¥–æ–∫

# –û–≥–æ–ª–æ—à–µ–Ω—å –∑–º—ñ–Ω–Ω–∏—Ö –Ω–µ–º–∞—î, —Ç—ñ–ª—å–∫–∏ –ø—Ä–∏—Å–≤–æ—î–Ω–Ω—è.
# –£–≥–æ–¥–∞ –ø—Ä–æ —ñ–º–µ–Ω—É–≤–∞–Ω–Ω—è –∑–º—ñ–Ω–Ω–∏—Ö - —Å—Ç–∏–ª—å snake_case
some_var = 5
some_var  # => 5

# –î–æ—Å—Ç—É–ø –¥–æ —Ä–∞–Ω—ñ—à–µ –Ω–µ –ø—Ä–∏—Å–≤–æ—î–Ω–æ—ó –∑–º—ñ–Ω–Ω–æ—ó —î –≤–∏–Ω—è—Ç–∫–æ–º.
# –î–∏–≤—ñ—Ç—å—Å—è Control Flow, —â–æ–± –¥—ñ–∑–Ω–∞—Ç–∏—Å—è –±—ñ–ª—å—à–µ –ø—Ä–æ –æ–±—Ä–æ–±–∫—É –≤–∏–Ω—è—Ç–∫—ñ–≤.
some_unknown_var  # –í–∏–∫–ª–∏–∫–∞—î NameError

# if –º–æ–∂–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏—Å—è —è–∫ –≤–∏—Ä–∞–∑
# –ï–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç —Ç–µ—Ä–Ω–∞—Ä–Ω–æ–≥–æ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ '?:' –≤ C
"yay!" if 0 > 1 else "nay!"  # => "nay!"

# –°–ø–∏—Å–∫–∏ –∑–±–µ—Ä—ñ–≥–∞—é—Ç—å –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ—Å—Ç—ñ
li = []
# –í–∏ –º–æ–∂–µ—Ç–µ –ø–æ—á–∞—Ç–∏ –∑ –∑–∞–ø–æ–≤–Ω–µ–Ω–æ–≥–æ —Å–ø–∏—Å–∫—É
other_li = [4, 5, 6]

# –î–æ–¥–∞–≤–∞–Ω–Ω—è –µ–ª–µ–º–µ–Ω—Ç—ñ–≤ –≤ –∫—ñ–Ω–µ—Ü—å —Å–ø–∏—Å–∫—É –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é append
li.append(1)    # li —Ç–µ–ø–µ—Ä [1]
li.append(2)    # li —Ç–µ–ø–µ—Ä [1, 2]
li.append(4)    # li —Ç–µ–ø–µ—Ä [1, 2, 4]
li.append(3)    # li —Ç–µ–ø–µ—Ä [1, 2, 4, 3]
# –í–∏–¥–∞–ª–µ–Ω–Ω—è –∑ –∫—ñ–Ω—Ü—è –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é pop
li.pop()        # => 3 —ñ li —Ç–µ–ø–µ—Ä [1, 2, 4]
# –î–∞–≤–∞–π—Ç–µ –ø–æ–≤–µ—Ä–Ω–µ–º–æ –π–æ–≥–æ –Ω–∞–∑–∞–¥
li.append(3)    # li –∑–Ω–æ–≤—É [1, 2, 4, 3]

# –î–æ—Å—Ç—É–ø –¥–æ —Å–ø–∏—Å–∫—É, —è–∫ –¥–æ –±—É–¥—å-—è–∫–æ–≥–æ –º–∞—Å–∏–≤—É
li[0]   # => 1
# –ü–æ–¥–∏–≤–∏—Ç–∏—Å—è –Ω–∞ –æ—Å—Ç–∞–Ω–Ω—ñ–π –µ–ª–µ–º–µ–Ω—Ç
li[-1]  # => 3

# –í–∏—Ö—ñ–¥ –∑–∞ –º–µ–∂—ñ –≤–∏–∫–ª–∏–∫–∞—î IndexError
li[4]  # –í–∏–∫–ª–∏–∫–∞—î IndexError

# –í–∏ –º–æ–∂–µ—Ç–µ –¥–∏–≤–∏—Ç–∏—Å—è –Ω–∞ –¥—ñ–∞–ø–∞–∑–æ–Ω–∏ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Å–∏–Ω—Ç–∞–∫—Å–∏—Å—É –∑—Ä—ñ–∑—ñ–≤.
# –ü–æ—á–∞—Ç–∫–æ–≤–∏–π —ñ–Ω–¥–µ–∫—Å –≤–∫–ª—é—á–µ–Ω–æ, –∫—ñ–Ω—Ü–µ–≤–∏–π - –Ω—ñ
# (–¶–µ –∑–∞–º–∫–Ω—É—Ç–∏–π/–≤—ñ–¥–∫—Ä–∏—Ç–∏–π –¥—ñ–∞–ø–∞–∑–æ–Ω –¥–ª—è –º–∞—Ç–µ–º–∞—Ç–∏–∫—ñ–≤.)

# –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ "def" –¥–ª—è —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è –Ω–æ–≤–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π

def varargs(*args):
    return args

varargs(1, 2, 3)  # => (1, 2, 3)

# –í–∏ –º–æ–∂–µ—Ç–µ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ —Ñ—É–Ω–∫—Ü—ñ—ó, —è–∫—ñ –ø—Ä–∏–π–º–∞—é—Ç—å –∑–º—ñ–Ω–Ω—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å
# —ñ–º–µ–Ω–æ–≤–∞–Ω–∏—Ö –∞—Ä–≥—É–º–µ–Ω—Ç—ñ–≤, —Ç–∞–∫–æ–∂
def keyword_args(**kwargs):
    return kwargs

# –î–∞–≤–∞–π—Ç–µ –≤–∏–∫–ª–∏—á–µ–º–æ —ó—ó, —â–æ–± –ø–æ–±–∞—á–∏—Ç–∏, —â–æ —Å—Ç–∞–Ω–µ—Ç—å—Å—è
keyword_args(big="foot", loch="ness")  # => {"big": "foot", "loch": "ness"}


# –í–∏ –º–æ–∂–µ—Ç–µ —Ä–æ–±–∏—Ç–∏ —ñ —Ç–µ, —ñ —ñ–Ω—à–µ –æ–¥–Ω–æ—á–∞—Å–Ω–æ, —è–∫—â–æ —Ö–æ—á–µ—Ç–µ
def all_the_args(*args, **kwargs):
    print(args)
    print(kwargs)
"""

    (1, 2)
    {"a": 3, "b": 4}
"""

# –ü—Ä–∏ –≤–∏–∫–ª–∏–∫—É —Ñ—É–Ω–∫—Ü—ñ–π –≤–∏ –º–æ–∂–µ—Ç–µ —Ä–æ–±–∏—Ç–∏ –ø—Ä–æ—Ç–∏–ª–µ–∂–Ω–µ –¥–æ args/kwargs!

args = (1, 2, 3, 4)
kwargs = {"a": 3, "b": 4}

all_the_args(**kwargs)         # –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω–æ: all_the_args(a=3, b=4)
all_the_args(*args, **kwargs)  # –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω–æ: all_the_args(1, 2, 3, 4, a=3, b=4)

# –ú–∏ –º–æ–∂–µ–º–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ –∫–ª—é—á–æ–≤–µ —Å–ª–æ–≤–æ nonlocal –¥–ª—è —Ä–æ–±–æ—Ç–∏ –∑—ñ –∑–º—ñ–Ω–Ω–∏–º–∏ —É –≤–∫–ª–∞–¥–µ–Ω—ñ–π –æ–±–ª–∞—Å—Ç—ñ –≤–∏–¥–∏–º–æ—Å—Ç—ñ, —è–∫—ñ –Ω–µ –ø–æ–≤–∏–Ω–Ω—ñ –±—É—Ç–∏ –æ–≥–æ–ª–æ—à–µ–Ω—ñ —É –≤–Ω—É—Ç—Ä—ñ—à–Ω—ñ—Ö —Ñ—É–Ω–∫—Ü—ñ—è—Ö.
def create_avg():
    total = 0
    count = 0
    def avg(n):
        nonlocal total, count
        total += n
        count += 1
        return total/count
    return avg
avg = create_avg()
avg(3)  # => 3.0
avg(5)  # (3+5)/2 => 4.0
avg(7)  # (8+7)/3 => 5.0

        # –ø—Ä–∞—Ü—é—î —Ç—ñ–ª—å–∫–∏ –∑ –Ω–∞—Å—Ç—É–ø–Ω–∏–º –±–∞–∑–æ–≤–∏–º –∫–ª–∞—Å–æ–º —É —Å–ø–∏—Å–∫—É MRO.
        # –¢–æ–º—É –∑–∞–º—ñ—Å—Ç—å —Ü—å–æ–≥–æ –º–∏ —è–≤–Ω–æ –≤–∏–∫–ª–∏–∫–∞—î–º–æ __init__ –¥–ª—è –≤—Å—ñ—Ö –ø—Ä–µ–¥–∫—ñ–≤.
        # –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è *args —Ç–∞ **kwargs –¥–æ–∑–≤–æ–ª—è—î —á–∏—Å—Ç–æ –ø–µ—Ä–µ–¥–∞–≤–∞—Ç–∏
        # –∞—Ä–≥—É–º–µ–Ω—Ç–∏, –ø—Ä–∏—á–æ–º—É –∫–æ–∂–µ–Ω –±–∞—Ç—å–∫–æ "–∑–Ω—ñ–º–∞—î —à–∞—Ä —Ü–∏–±—É–ª—ñ".
        Superhero.__init__(self, "anonymous", movie=True,

# –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä–∏ –¥–æ–ø–æ–º–∞–≥–∞—é—Ç—å —Å—Ç–≤–æ—Ä—é–≤–∞—Ç–∏ "–ª—ñ–Ω–∏–≤–∏–π" –∫–æ–¥.
def double_numbers(iterable):
    for i in iterable:
        yield i + i

# –ì–µ–Ω–µ—Ä–∞—Ç–æ—Ä–∏ –µ—Ñ–µ–∫—Ç–∏–≤–Ω—ñ –∑ —Ç–æ—á–∫–∏ –∑–æ—Ä—É –ø–∞–º'—è—Ç—ñ, –æ—Å–∫—ñ–ª—å–∫–∏ –≤–æ–Ω–∏ –∑–∞–≤–∞–Ω—Ç–∞–∂—É—é—Ç—å –ª–∏—à–µ –¥–∞–Ω—ñ, –Ω–µ–æ–±—Ö—ñ–¥–Ω—ñ –¥–ª—è
# –æ–±—Ä–æ–±–∫–∏ –Ω–∞—Å—Ç—É–ø–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è –≤ —ñ—Ç–µ—Ä–∞–±–µ–ª—å–Ω–æ–º—É –æ–±'—î–∫—Ç—ñ. –¶–µ –¥–æ–∑–≤–æ–ª—è—î —ó–º –≤–∏–∫–æ–Ω—É–≤–∞—Ç–∏
# –æ–ø–µ—Ä–∞—Ü—ñ—ó –Ω–∞–¥ —ñ–Ω–∞–∫—à–µ –∑–∞–Ω–∞–¥—Ç–æ –≤–µ–ª–∏–∫–∏–º–∏ –¥—ñ–∞–ø–∞–∑–æ–Ω–∞–º–∏ –∑–Ω–∞—á–µ–Ω—å.
# –ü–†–ò–ú–Ü–¢–ö–ê: `range` –∑–∞–º—ñ–Ω—é—î `xrange` –≤ Python 3.

    print(i)
    if i >= 30:
        break
```
